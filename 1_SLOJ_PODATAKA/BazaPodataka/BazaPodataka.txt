/****** Object:  View [dbo].[vw_poreska_prijava_detalji]    Script Date: 6/1/2026 7:25:56 PM ******/

GO
SET QUOTED_IDENTIFIER ON

CREATE VIEW [dbo].[vw_poreska_prijava_detalji] AS
SELECT 
    pp.id AS poreska_prijava_id,
    l_obv.AdresaId,

    -- OBVEZNIK
    pp.obveznik_lice_id,
    COALESCE(NULLIF(CONCAT(l_obv.ime, ' ', l_obv.prezime), ' '), l_obv.naziv) AS obveznik_lice_naziv,

    l_obv.jmbg,
    l_obv.pib,
    l_obv.telefon,
    korisnik.email,

    -- ADRESA OBVEZNIKA
    a.sprat AS obveznik_sprat,
    a.stan AS obveznik_stan,
    a.ulicaIBroj AS obveznik_ulica_broj,
    m.naziv AS obveznik_mesto,
    m.ptt AS obveznik_ptt,
    

    -- PODNOSILAC
    pp.podnosilac_lice_id,
    COALESCE(NULLIF(CONCAT(l_pod.ime, ' ', l_pod.prezime), ' '), l_pod.naziv) AS podnosilac_lice_naziv,

    pp.je_izmenjena,
    pp.vrsta_prava_id,
    CONCAT(vp.sifra, ' - ', vp.naziv) AS vrsta_prava_naziv,
    vp.sifra,
    pp.jls_id,
    jls.naziv AS jls_naziv,

    pp.dan_nastanka_poreske_obaveze,
    pp.dan_prestanka_poreske_obaveze,
    pp.razlog_prestanka_id,


    pp.ima_poreski_kredit AS pp_ima_poreski_kredit,
    pp.kredit_osnov AS pp_kredit_osnov,

    pp.ima_poresko_oslobodjenje AS pp_ima_poresko_oslobodjenje,
    pp.oslobodjenje_osnov AS pp_oslobodjenje_osnov,

    pp.datum_podnosenja,
    pp.napomena AS pp_napomena,
    pp.kreirana_u,
    pp.azurirana_u,
    pp.popis_prilozenih_dokaza,
    pp.OrganizacionaJedinica,

    -- NEPOKRETNOST
    n.id AS nepokretnost_id,
    n.prijava_id AS nepokretnost_prijava_id,
    n.vrsta_id AS nepokretnost_vrsta_id,
    n.rbr,

    n.katastarska_opstina_id,
    ko.naziv AS katastarska_opstina_naziv,

    n.UlicaIbroj,
    n.br_kat_parcele,
    n.godina_izgradnje,
    n.korisna_povrsina,
    n.ukupna_povrsina,
    n.udeo_obveznika,

    n.ima_poreski_kredit AS n_ima_poreski_kredit,
    n.ima_poresko_oslobodjenje AS n_ima_poresko_oslobodjenje,
    n.oslobodjenje_osnov AS n_oslobodjenje_osnov,

    n.napomena AS n_napomena

FROM poreska_prijava pp

LEFT JOIN Nepokretnost n 
    ON n.prijava_id = pp.id

LEFT JOIN Lice l_obv 
    ON l_obv.id = pp.obveznik_lice_id

LEFT JOIN Adresa a 
    ON a.id = l_obv.AdresaId

LEFT JOIN Mesto m 
    ON m.id = a.mesto_id

LEFT JOIN Lice l_pod 
    ON l_pod.id = pp.podnosilac_lice_id

LEFT JOIN vrsta_prava vp 
    ON vp.id = pp.vrsta_prava_id

LEFT JOIN jls 
    ON jls.id = pp.jls_id
LEFT JOIN korisnik  
ON l_pod.id = korisnik.LiceID
LEFT JOIN katastarska_opstina ko 
    ON ko.id = n.katastarska_opstina_id;
